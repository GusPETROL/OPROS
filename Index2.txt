<script>
    const form = document.getElementById('quizForm');
    const submitBtn = document.getElementById('submitBtn');
    const resultBox = document.getElementById('resultBox');
    const errorMessage = document.getElementById('errorMessage');
    const resetBtn = document.getElementById('resetBtn');

    function getSelectedValue(name) {
        const selected = document.querySelector(`input[name="${name}"]:checked`);
        return selected ? parseInt(selected.value) : null;
    }

    function calculateResult() {
        const answers = {
            q1: getSelectedValue('q1'),
            q2: getSelectedValue('q2'),
            q3: getSelectedValue('q3'),
            q4: getSelectedValue('q4'),
            q5: getSelectedValue('q5')
        };

        // Проверка что все вопросы отвечены
        for (let key in answers) {
            if (answers[key] === null) {
                return null;
            }
        }

        const totalScore = answers.q1 + answers.q2 + answers.q3 + answers.q4 + answers.q5;
        return totalScore;
    }

    function showResult(score) {
        form.style.display = 'none';
        resultBox.style.display = 'block';
        errorMessage.style.display = 'none';

        const titleElem = document.getElementById('resultTitle');
        const textElem = document.getElementById('resultText');

        if (score <= 2) {
            titleElem.textContent = 'Нужно больше узнать';
            textElem.textContent = 'Ты пока плохо понимаешь, что такое кибербуллинг и как с ним бороться. Внимательно изучи материалы на сайте — это поможет защитить себя и друзей.';
        } else if (score <= 4) {
            titleElem.textContent = 'Хороший результат, но есть над чем работать';
            textElem.textContent = 'Ты уже многое знаешь, но некоторые важные моменты упускаешь. Прочитай раздел «Что делать при кибербуллинге», чтобы стать увереннее.';
        } else {
            titleElem.textContent = 'Отличный результат';
            textElem.textContent = 'Ты отлично разбираешься в теме кибербуллинга. Знаешь, как распознать травлю и правильно действовать. Поделись этими знаниями с друзьями.';
        }
    }

    submitBtn.addEventListener('click', () => {
        const score = calculateResult();
        if (score === null) {
            errorMessage.style.display = 'block';
            return;
        }
        showResult(score);
    });

    resetBtn.addEventListener('click', () => {
        // Сбросить все радиокнопки
        const radioButtons = document.querySelectorAll('input[type="radio"]');
        radioButtons.forEach(radio => radio.checked = false);
        form.style.display = 'block';
        resultBox.style.display = 'none';
        errorMessage.style.display = 'none';
    });
</script>
